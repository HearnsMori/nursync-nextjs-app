//For get method
import { fetchData } from '../../utils/fetchdata'; // adjust path as needed

// GET example (load a single user)
interface User { id: string; name: string; email: string; }

(async () => {
  try {
    const user = await fetchData<User>({
      jsonData: null,
      backendURL: 'https://api.example.com',
      apiEndPoint: 'users/123',
      method: 'GET',
    });
    console.log('user', user);
  } catch (err) {
    console.error('GET failed', err);
  }
})();

//For post method
// POST example (create a user)
interface CreateUserPayload { name: string; email: string; password: string; }
interface CreateUserResponse { id: string; createdAt: string; }

const payload: CreateUserPayload = {
  name: 'Alice',
  email: 'alice@example.com',
  password: 'supersecret',
};

(async () => {
  try {
    const res = await fetchData<CreateUserResponse>({
      jsonData: payload,
      backendURL: 'https://api.example.com',
      apiEndPoint: 'users',
      method: 'POST',
    });
    console.log('created user', res);
  } catch (err) {
    console.error('POST failed', err);
  }
})();
